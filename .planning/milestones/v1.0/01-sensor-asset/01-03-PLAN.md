       ```dart
       /// Live sensor widget driven by a bool detection state key.
       ///
       /// Subscribes to `config.detectionKey` via `stateManProvider`. Stream is
       /// hoisted to `initState` (Pitfall 2). Visual flips immediately on bool
       /// change (no animation, no smoothing — SENS-05). Renders neutral grey
       /// when the key is empty, the stream has no value yet, or the stream
       /// errored (SENS-14). Honours `Coordinates.angle` via `LayoutRotatedBox`.
       /// Tap opens the config dialog through a real `GestureDetector` so the
       /// gesture survives a translating parent (Phase 3 forward-compat).
       class Sensor extends ConsumerStatefulWidget {
         final SensorConfig config;
         const Sensor({super.key, required this.config});

         @override
         ConsumerState<Sensor> createState() => _SensorState();
       }

       class _SensorState extends ConsumerState<Sensor> {
         Stream<bool>? _detectionStream;

         @override
         void initState() {
           super.initState();
           _hoistStream();
         }

         @override
         void didUpdateWidget(covariant Sensor oldWidget) {
           super.didUpdateWidget(oldWidget);
           if (oldWidget.config.detectionKey != widget.config.detectionKey) {
             _hoistStream();
           }
         }

         /// Construct the bool stream once (in initState or when the key changes).
         /// Pitfall 2: never inline this in build().
         void _hoistStream() {
           final key = widget.config.detectionKey;
           if (key.isEmpty) {
             _detectionStream = null;
             return;
           }
           _detectionStream = ref
               .read(stateManProvider.future)
               .asStream()
               .asyncExpand((sm) => sm.subscribe(key).asStream())
               .asyncExpand((s) => s)
               .map((dv) => dv.asBool);
         }

         CustomPainter _createPainter({required bool isActive, required bool isStale}) {
           switch (widget.config.kind) {
             case SensorKind.redLight:
               return RedLightBeamPainter(
                 isActive: isActive,
                 activeColor: widget.config.activeColor,
                 inactiveColor: widget.config.inactiveColor,
                 isStale: isStale,
               );
             case SensorKind.opticField:
               return OpticFieldPainter(
                 isActive: isActive,
                 activeColor: widget.config.activeColor,
                 inactiveColor: widget.config.inactiveColor,
                 isStale: isStale,
               );
             case SensorKind.inductiveField:
               return InductiveFieldPainter(
                 isActive: isActive,
                 activeColor: widget.config.activeColor,
                 inactiveColor: widget.config.inactiveColor,
                 isStale: isStale,
               );
           }
         }

         void _openConfigDialog(BuildContext context) {
           showDialog<void>(
             context: context,
             builder: (_) => widget.config.configure(context),
           );
         }

         Widget _buildPaint(CustomPainter painter) {
           final angleDeg = widget.config.coordinates.angle ?? 0.0;
           return LayoutRotatedBox(
             angle: angleDeg * pi / 180,
             child: GestureDetector(
               behavior: HitTestBehavior.opaque,
               onTap: () => _openConfigDialog(context),
               child: CustomPaint(painter: painter),
             ),
           );
         }

         @override
         Widget build(BuildContext context) {
           // Stale path #1: empty key — no stream constructed.
           if (_detectionStream == null) {
             return _buildPaint(_createPainter(isActive: false, isStale: true));
           }

           return StreamBuilder<bool>(
             stream: _detectionStream,
             builder: (context, snapshot) {
               // Stale path #2: stream emitted nothing yet, or errored.
               if (!snapshot.hasData || snapshot.hasError) {
                 return _buildPaint(_createPainter(isActive: false, isStale: true));
               }
               final isActive = sensorIsActive(
                 rawBool: snapshot.data!,
                 invertActivePolarity: widget.config.invertActivePolarity,
               );
               return _buildPaint(_createPainter(isActive: isActive, isStale: false));
             },
           );
         }
       }
       ```

    Note on Pitfall 2 wiring: `ref.read(stateManProvider.future)` returns a `Future<StateMan>`. Converting to a stream via `.asStream().asyncExpand((sm) => sm.subscribe(key).asStream()).asyncExpand((s) => s)` is the documented pattern that doesn't trigger `ref.watch`-style cascade invalidation. The resulting `Stream<bool>` is stored in `_detectionStream`. `StreamBuilder` receives the SAME stream instance across rebuilds — so `identical(oldStream, newStream)` is true → no resubscribe.

    Note on tap-through-translate: GestureDetector with `behavior: HitTestBehavior.opaque` AND living inside `Transform.translate` (named constructor — `transformHitTests: true` by default) means hit tests are forwarded correctly. UI-SPEC §Interaction Contract bans the raw `Transform()` constructor — using `Transform.translate(...)` in tests is correct.
