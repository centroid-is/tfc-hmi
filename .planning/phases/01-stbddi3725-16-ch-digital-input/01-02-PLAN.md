`_combinedStream` (beckhoff.dart:1272-1288):
```dart
CombineLatestStream<DynamicValue, Map<String, DynamicValue>> _combinedStream(
    LinkedHashMap<String, String?> keys, StateMan stateMan) {
  return CombineLatestStream([
    for (var entry in keys.entries)
      if (entry.value != null)
        stateMan.subscribe(entry.value!).asStream().asyncExpand((s) => s),
  ], (values) {
    final map = <String, DynamicValue>{};
    int i = 0;
    for (var entry in keys.entries) {
      if (entry.value != null) map[entry.key] = values[i++];
    }
    return map;
  });
}
```

    class _STBDDI3725State extends ConsumerState<_STBDDI3725> {
      late final Future<StateMan> _stateManFuture;
      Stream<Map<String, DynamicValue>>? _combinedStreamCache;  // hoisted (QUAL-03)
      StateMan? _stateMan;

      @override
      void initState() {
        super.initState();
        _stateManFuture = ref.read(stateManProvider.future);
        _stateManFuture.then((sm) {
          if (!mounted) return;
          setState(() {
            _stateMan = sm;
            _combinedStreamCache = _combinedStream(
              LinkedHashMap.fromEntries([
                MapEntry('raw', widget.config.rawStateKey),
                MapEntry('force', widget.config.forceValuesKey),
              ]),
              sm,
            );
          });
        });
      }

      @override
      Widget build(BuildContext context) {
        if (_combinedStreamCache == null) {
          return _build(ledStates: List.filled(16, IOState.low), isStale: true);
        }
        return StreamBuilder<Map<String, DynamicValue>>(
          stream: _combinedStreamCache,
          builder: (context, snap) {
            final data = (snap.hasData && !snap.hasError) ? snap.data : null;
            if (data == null) return _build(ledStates: List.filled(16, IOState.low), isStale: true);
            final raw = data['raw']?.asInt ?? 0;
            final forceList = _forceArrayFromDynamicValue(data['force']);
            final leds = bitmaskToLedStates(raw, forceValues: forceList);
            return _build(ledStates: leds, isStale: false);
          },
        );
      }

      Widget _build({required List<IOState> ledStates, required bool isStale}) {
        return GestureDetector(
          behavior: HitTestBehavior.opaque,
          onTap: () {
            // Plan 03 will replace this with the detail dialog.

              builder: (_) => const AlertDialog(

          child: STBDDI3725Widget(
            ledStates: ledStates,
            isStale: isStale,
            isDisconnected: false,
            animation: const AlwaysStoppedAnimation(0),
          ),
        );
      }
    }
