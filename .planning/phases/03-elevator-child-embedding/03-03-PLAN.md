    Add three private handler methods to `_ElevatorConfigEditorState`:

    ```dart
    /// Hard-coded list per CONTEXT §specifics + ELEV-07. The user explicitly
    /// locked the dropdown to {Sensor, Conveyor} — do NOT iterate
    /// AssetRegistry.defaultFactories and filter; the explicit list is the
    /// safer expression of the locked decision.
    static final Map<String, BaseAsset Function()> _allowedChildFactories = {
      'Sensor': SensorConfig.preview,
      'Conveyor': ConveyorConfig.preview,
    };

    Future<void> _onAddChildPressed(BuildContext context) async {
      final selected = await showMenu<String>(
        context: context,
        position: const RelativeRect.fromLTRB(100, 100, 100, 100),
        items: _allowedChildFactories.keys.map((name) =>
          PopupMenuItem<String>(value: name, child: Text(name))
        ).toList(),
      );
      if (selected == null) return;
      final factory = _allowedChildFactories[selected]!;
      setState(() {
        widget.config.children.add(ElevatorChildEntry(
          // id auto-generated by ElevatorChildEntry constructor (microsecondsSinceEpoch)
          offsetX: 0.5,
          child: factory(),
        ));
      });
    }

    void _onEditChildPressed(BuildContext context, ElevatorChildEntry entry) {
      showDialog<void>(
        context: context,
        builder: (_) => Dialog(
          child: entry.child.configure(context),
        ),
      ).then((_) => setState(() {}));
    }

    void _onRemoveChildPressed(ElevatorChildEntry entry) {
      setState(() {
        widget.config.children.removeWhere((e) => e.id == entry.id);
      });
    }
    ```
