    ```dart
    /// Tooltip content for the Sensor asset.
    ///
    /// Subscribes to rising/falling edge-delay state keys ONLY while this widget
    /// is mounted (Flutter mounts tooltip content widgets when the tooltip opens
    /// and unmounts them on dismiss — this satisfies the locked subscribe-on-open,
    /// cancel-on-close contract from 01-CONTEXT.md without explicit subscription
    /// bookkeeping).
    class _SensorTooltipContent extends ConsumerWidget {
      final SensorConfig config;
      const _SensorTooltipContent({required this.config});

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        // Stale paths first — UI-SPEC copy contract.
        if (config.detectionKey.isEmpty) {
          return const Text('Detection key not set');
        }

        // Build a 2-line tooltip: rising and falling.
        return DefaultTextStyle(
          style: Theme.of(context).textTheme.bodySmall ??
              const TextStyle(color: Colors.white, fontSize: 12),
          child: Column(
            mainAxisSize: MainAxisSize.min,
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              _DelayRow(label: 'Rising',  key: config.risingEdgeDelayKey),
              _DelayRow(label: 'Falling', key: config.fallingEdgeDelayKey),
            ],
          ),
        );
      }
    }

    /// Single-line "Rising: <ms>ms" or "Rising: —" or "Rising: …" row.
    class _DelayRow extends ConsumerWidget {
      final String label;
      // ignore: avoid_renaming_method_parameters — `key` is the state-key string here, NOT a Widget Key.
      final String key;

      // Use a different parameter name on the constructor to avoid shadowing `super.key`.
      const _DelayRow({required this.label, required String key}) : key = key;

      @override
      Widget build(BuildContext context, WidgetRef ref) {
        if (key.isEmpty) {
          return Text('$label: —');
        }
        return StreamBuilder<num>(
          stream: ref.read(stateManProvider.future).asStream().asyncExpand(
                (sm) => sm.subscribe(key).asStream().asyncExpand((s) => s),
              ).map((dv) => dv.asInt),
          builder: (context, snapshot) {
            if (snapshot.hasError) {
              return Text('$label: —');
            }
            if (!snapshot.hasData) {
              return Text('$label: …');
            }
            return Text('$label: ${snapshot.data}ms');
          },
        );
      }
    }
    ```
