```dart
class MSocket {
  final _status = BehaviorSubject<ConnectionStatus>.seeded(ConnectionStatus.disconnected);
  static const _initialBackoff = Duration(milliseconds: 500);
  static const _maxBackoff = Duration(seconds: 5);
  Duration _backoff = _initialBackoff;
  bool _disposed = false;

  void connect() { _disposed = false; _connectionLoop(); }
  void dispose() { _disposed = true; _destroySocket(); _dataController.close(); _status.close(); }

  Future<void> _connectionLoop() async {
    while (!_disposed) {
      if (!_status.isClosed) _status.add(ConnectionStatus.connecting);
      try {
        _socket = await Socket.connect(host, port, timeout: const Duration(seconds: 3));
        if (_disposed) { _destroySocket(); break; }
        // ... setup ...
        if (!_status.isClosed) _status.add(ConnectionStatus.connected);
        _backoff = _initialBackoff;
        await done.future; // block until disconnect
      } catch (e) { ... }
      _destroySocket();
      if (!_disposed && !_status.isClosed) _status.add(ConnectionStatus.disconnected);
      if (_disposed) break;
      await Future.delayed(_backoff);
      if (_disposed) break;
      _backoff = _clampDuration(_backoff * 2, Duration.zero, _maxBackoff);
    }
  }
}
```

    ```dart
    import 'dart:async';
    import 'package:modbus_client/modbus_client.dart';
    import 'package:modbus_client_tcp/modbus_client_tcp.dart';
    import 'package:rxdart/rxdart.dart';
    import 'state_man.dart' show ConnectionStatus;
    // Use logger if helpful: import 'package:logger/logger.dart';

    class ModbusClientWrapper {
      final String host;
      final int port;
      final int unitId;
      final ModbusClientTcp Function(String host, int port, int unitId) _clientFactory;

      final _status = BehaviorSubject<ConnectionStatus>.seeded(ConnectionStatus.disconnected);
      ModbusClientTcp? _client;
      bool _stopped = true;
      bool _disposed = false;

      static const _initialBackoff = Duration(milliseconds: 500);
      static const _maxBackoff = Duration(seconds: 5);
      Duration _backoff = _initialBackoff;

      // Constructor with factory injection
      ModbusClientWrapper(this.host, this.port, this.unitId, {
        ModbusClientTcp Function(String, int, int)? clientFactory,
      }) : _clientFactory = clientFactory ?? _defaultFactory;

      static ModbusClientTcp _defaultFactory(String host, int port, int unitId) {
        return ModbusClientTcp(host,
          serverPort: port,
          unitId: unitId,
          connectionMode: ModbusConnectionMode.doNotConnect,
          connectionTimeout: const Duration(seconds: 3),
        );
      }

      // Public API
      ConnectionStatus get connectionStatus => _status.value;
      Stream<ConnectionStatus> get connectionStream => _status.stream;

      void connect() { _stopped = false; _disposed = false; _connectionLoop(); }
      void disconnect() { _stopped = true; _cleanupClient(); }
      void dispose() { _stopped = true; _disposed = true; _cleanupClient(); _status.close(); }
    }
    ```

    **Connection loop** -- follow MSocket's `_connectionLoop()` pattern:
    - `while (!_stopped)` loop
    - Emit `connecting` before each attempt
    - Create client via factory, call `connect()`
    - Check `_stopped` after await (race condition guard per Pitfall 3)
    - On success: emit `connected`, reset `_backoff`, poll `isConnected` every 250-500ms
    - On failure/disconnect: emit `disconnected`, apply backoff delay, double backoff (clamped to 5s max)
    - Check `_stopped` after every await and before every `_status.add()`
    - Always guard `_status.add()` with `!_status.isClosed` check (per Pitfall 4)

    **Disconnect detection** -- poll `_client.isConnected` every ~250ms after successful connect:
    ```dart
    // After connect succeeds:
    while (!_stopped && _client != null && _client!.isConnected) {
      await Future.delayed(const Duration(milliseconds: 250));
    }
    ```

    **Cleanup helper:**
    ```dart
    Future<void> _cleanupClient() async {
      final client = _client;
      _client = null;
      if (client != null) await client.disconnect();
      if (!_stopped || !_status.isClosed) {
        // Only emit disconnected if not already
        if (_status.value != ConnectionStatus.disconnected && !_status.isClosed) {
          _status.add(ConnectionStatus.disconnected);
        }
      }
    }
    ```
